def mean_error(actual,predicted):
    error=actual-predicted
    return error.mean() 

def results(data):
    print('RMSE:',sqrt(mean_squared_error(data.Actuals,data.Predictions)))
    print('MEAN_ERROR:',mean_error(data.Actuals,data.Predictions))
    print('MEAN_ABSOLUTE_ERROR:',mean_absolute_error(data.Actuals,data.Predictions))
    print('Corelation Between "Actual_Sales" and "Predicted_Sales":',data['Actuals'].corr(data['Predictions']))
    
def MAPE(actual,predicted):
    mape = ((predicted - actual)/actual)*100
    return mape
